5) Code Walkthrough (Conceptual Example)

This mirrors the solution approach shown in the transcript. You can adapt naming and markup to your starter project.

// Conceptual React code (use inside your React project)
import { useState } from "react";

function Counter() {
  const [count, setCount] = useState(0);
  const [step, setStep] = useState(1);

  // Base date can be today OR a fixed date (useful for demos)
  const date = new Date();
  // Example fixed base date (optional): new Date("June 21 2027")

  // Move the date by count days
  date.setDate(date.getDate() + count);

  function countLabel() {
    if (count === 0) return "Today is";
    if (count > 0) return `${count} day${count === 1 ? "" : "s"} from now is`;
    return `${Math.abs(count)} day${Math.abs(count) === 1 ? "" : "s"} ago was`;
  }

  return (
    <div>
      <div>
        <button onClick={() => setStep(s => s - 1)}>âˆ’</button>
        <span> Step: {step} </span>
        <button onClick={() => setStep(s => s + 1)}>+</button>
      </div>

      <div>
        <button onClick={() => setCount(c => c - step)}>âˆ’</button>
        <span> Count: {count} </span>
        <button onClick={() => setCount(c => c + step)}>+</button>
      </div>

      <p>
        <span>{countLabel()}</span> {date.toDateString()}
      </p>
    </div>
  );
}

export default function App() {
  return (
    <div>
      <Counter />
    </div>
  );
}

Why the callback form is used (important) ğŸ”’
